{Apple Strudel Muffins}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    %\introduction{

    %}
    
    \ingredients{%
        Muffin:
        2 cups & Flour \\
        1 tsp & Baking Powder \\
        1/2 tsp & Baking Soda \\
        1/2 cup & Butter \\
        1 cup & White Sugar \\
        2 & eggs \\
        1 1/4 tsp & Vanilla \\
        1 1/2 cups & Apples (diced) \\
        Strudel:
        1/3 cup & Brown Sugar \\
        1 Tbls & Flour \\
        1/8 tsp & Cinnamon \\
        1 Tbls & Butter \\
    }
    
    \preparation{%
        \step Mix flour, baking powder and baking soda.
        \step Beat butter, sugar, and eggs.  Add vanilla.
        \step Mix in apples.
        \step Incorporate dry ingredients.
        \step Scoop into muffin pan.
        \step mix brown sugar, flour, and cinnamon in small bowl.  Cut in butter.
        \step Bake, and let cool before devouring
    }

    %}
    
    \hint{%
        Once adding in dry ingredients, do not overmix.
    }
